sga36 <- kynba36 %>% filter(Player == "Shai Gilgeous-Alexander")

db36 <- kynba36 %>% filter(Player == "Devin Booker")
ad36 <- kynba36 %>% filter(Player == "Anthony Davis")
jw36 <- kynba36 %>% filter(Player == "John Wall")

kytop5pts <- kynba36 %>% top_n(5, wt=PTS)
bar1 <- ggplot() +
  geom_bar(data=kytop5pts, aes(x=reorder(Player, PTS), weight=PTS)) +
  geom_bar(data=sga36,aes(x=reorder(Player, PTS), weight=PTS), fill="#007AC1") +
  geom_bar(data=df36,aes(x=reorder(Player, PTS), weight=PTS), fill="#5A2D81") +
  geom_bar(data=db36,aes(x=reorder(Player, PTS), weight=PTS), fill="#E56020") +
  geom_bar(data=ad36,aes(x=reorder(Player, PTS), weight=PTS), fill="#FDB927") +
  coord_flip() +

    x="",
    y="Points per 36 minutes",
    title="Shai is the only player who is top 5 in all categories",

table <- kynba100 %>%
  filter(MP/G > 20) %>%
  mutate(NRtg = ORtg-DRtg) %>%
  select(Player, NRtg, ORtg, DRtg, PTS) %>%
  arrange(desc(NRtg))

table %>% 
  gt() %>%
  cols_label(
    Player = "Player",
    PTS = "Points",
    ORtg = "Offensive Rating",
    DRtg = "Defensive Rating",
    NRtg = "Net Rating"
  ) %>%
  tab_header(
    title = "Davis and Gilgeous-Alexander lead in Net Rating and are top 5 in Points",
    subtitle = "Shai, once again, is a leader in offensive and defensive ratings compared to the other Kentucky players.") %>%

    style = cell_text(color = "black", weight = "bold", align = "left"),
    locations = cells_title("title")
